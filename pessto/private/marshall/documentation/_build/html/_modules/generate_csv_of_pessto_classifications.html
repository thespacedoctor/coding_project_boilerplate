[docs]def grab_coordinates(transientBucketId):

	# CONNECT TO DATABASE
	db = pm.connect_to_database()

	# CREATE THE QUERY OBJECT
	query = "SELECT DISTINCT raDeg, decDeg \
	FROM transientBucket WHERE transientBucketId = "+str(transientBucketId)+" AND raDeg is not NULL AND decDeg is not NULL"

	# MAKE THE DATABASE CURSOR
	cursor=db.cursor(ms.cursors.DictCursor)

	try:
		# EXECUTE THE SQL QUERY
		cursor.execute(query)
	except ms.Error, e:
		errorString = "Something went wrong. Here's the mysql error:"+str(e)
		print errorString
		sys.exit()

	# COLLECT AND RETURN THE RESULTS - A LIST OF DICTIONARIES
	rows =  cursor.fetchall()
	raDeg, decDeg = rows[0]["raDeg"], rows[0]["decDeg"]
	return raDeg, decDeg
