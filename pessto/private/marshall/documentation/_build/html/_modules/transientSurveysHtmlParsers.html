[docs]def parse_ogle_pages(downloadDirectory,dbConn,log):
  """
  Downloads the ogle page: http://ogle.astrouw.edu.pl/ogle4/transients/transients.html and parses the transient metadata into db

  Key Arguments:
    - downloadDirectory -- directory to download the OGLE data files
    - dbConn -- mysql database connection
    - log -- logger

  Returns
    - None

  created: November 12, 2012
  last modified: November 13, 2012
  author: dryx
  status: tidy and complete as of November 13, 2012
  """
  ################ > IMPORTS ################
  import dryxWebCrawlers as wc
  import re
  import utils as u
  import dryxCommonUtils as b
  import dryxMySQL as m

  ################ >SETTINGS ################
  url = "http://ogle.astrouw.edu.pl/ogle4/transients/transients.html"
  urlp = "http://ogle.astrouw.edu.pl/ogle4/transients/"

  sumRow = {}
  recRow = []

  # REGEXS
  reAllData = re.compile(r'<hr>.*<hr>', re.DOTALL)
  reDataList = re.compile(r'(?=(<hr>.*?<hr>))',re.S)
  reLcDataFile = re.compile(r'images.*?dat')
  reLcPlot = re.compile(r'images.*?plot.*?png')
  reRefImg = re.compile(r'images.*?chart.*?gif')
  reRefFits = re.compile(r'images.*?chart.*?fts')
  reTargetImg = re.compile(r'images.*?frame.*?gif')
  reTargetFits = re.compile(r'images.*?frame.*?fts')
  reSubImg = re.compile(r'images.*?sub.*?gif')
  reSubFits = re.compile(r'images.*?sub.*?fts')
  reName = re.compile(r'OGLE-\d{4}-[a-zA-Z]{2,4}-\d{3}')
  reTypeA = re.compile(r'-[a-zA-Z]{2,4}-')
  reTypeB = re.compile(r'[a-zA-Z]{2,4}')
  reRaDec = re.compile(r'\d{1,2}:.*')
  reJD = re.compile(r'24.*?\s')
  reMag = re.compile(r'\s\d{1,2}\.\d{1,4}\s')

  ## LAMBDA FUNCTIONS ##
  f = lambda x: x.search(data.group(1)).group()
  fl = lambda x: x.search(line).group()

  ################ >ACTION(S) ################
  try:
    log.debug("attempting to download the ogle transient webpage")
    localFile = wc.singleWebDocumentDownloader(url,downloadDirectory,log,dbConn,1)
  except Exception, e:
    log.error("could not download the ogle transient webpage - failed with this error %s: " % (str(e),))
    return -1

  ## READ INTO THE DATABLOCK FROM THE WEBPAGE
  rf=open(localFile,'r')
  allData = rf.read()

  try:
    log.debug("attempting to read the data from the ogle html page")
    allData = reAllData.search(allData).group()
  except Exception, e:
    log.error("could not read the data from the ogle html page - failed with this error %s: " % (str(e),))
    return -1

  ## SPLIT THE DATA INTO SINGLE OBJECTS & ITERATE
  try:
    log.debug("attempting to split the ogle data into seperate object blocks")
    dataList = reDataList.finditer(allData)
    #dataList = [match.group(1) for match in matches]
  except Exception, e:
    log.error("could not split the ogle data into seperate object blocks - failed with this error: %s " % (str(e),))
    return -1

  count = 0
  for data in dataList:
    count += 1
    sumRow = {}
    recRow = []
    ##log.debug('one object block:\n %s' % (data.group(),))
    ## GRAB THE SUMMARY (DISCOVERY) INFO FOR THE OGLE TRANSIENT
    try:
      log.debug("attempting to grab an ogle lightcurve plot")
      lcPlotUrl = urlp+f(reLcPlot)
      sumRow['lightcurveUrl'] = lcPlotUrl
    except Exception, e:
      log.error("could not grab an ogle lightcurve plot - failed with this error %s: " % (str(e),))

    try:
      log.debug("attempting to download an ogle reference image")
      refImgUrl = urlp+f(reRefImg)
      refFitsUrl = urlp+f(reRefFits)
      #log.debug("reference image url: "+refImgUrl)
      sumRow['referenceImageUrl'] = refImgUrl
      sumRow['referenceFitsUrl'] = refFitsUrl
    except Exception, e:
      log.error("could not download an ogle reference image - failed with this error %s: " % (str(e),))

    try:
      log.debug("attempting to download an ogle target image")
      targetImgUrl = urlp+f(reTargetImg)
      targetFitsUrl = urlp+f(reTargetFits)
      sumRow['targetImageUrl'] = targetImgUrl
      sumRow['targetFitsUrl'] = targetFitsUrl
    except Exception, e:
      log.error("could not download an ogle target image - failed with this error %s: " % (str(e),))

    try:
      log.debug("attempting to download an ogle subtraction image")
      subImgUrl = urlp+f(reSubImg)
      subFitsUrl = urlp+f(reSubFits)
      sumRow['subtractedImageUrl'] = subImgUrl
      sumRow['subtractedFitsUrl'] = subFitsUrl
    except Exception, e:
      log.error("could not download an ogle subtraction image - failed with this error %s: " % (str(e),))

    try:
      log.debug("attempting to parse ogle name")
      name = f(reName)
      oType = reTypeB.search(reTypeA.search(name).group()).group()
      sumRow['name'] = name
      sumRow['transientTypePrediction'] = oType
    except Exception, e:
      log.error("could not parse ogle name - failed with this error %s: " % (str(e),))
      return -1

    try:
      log.debug("attempting to paser ogle object coordinates")
      raDec = f(reRaDec)
      raDecList = raDec.split(" ")
      ra, dec = raDecList[0], raDecList[1]
      sumRow['raDeg'] =u.sexToDec(ra,ra=True)
      sumRow['decDeg'] =u.sexToDec(dec,ra=False)
    except Exception, e:
      log.error("could not paser ogle object coordinates - failed with this error %s: " % (str(e),))
      return -1

    ## GRAB THE RECURRENCE EVENTS
    lcDataFileUrl = urlp+f(reLcDataFile)
    try:
      log.debug("attempting to download a datafile for an ogle object")
      lcDataFile = wc.singleWebDocumentDownloader(lcDataFileUrl,downloadDirectory,log,dbConn,1)
      rf2=open(lcDataFile,'r')
    except Exception, e:
      log.error("could not download a datafile for an ogle object - failed with this error %s: " % (str(e),))
      return -1

    lndSwitch = 0
    ndMJD = None
    ndDate = None
    sumRow['lastNonDetectionDate'] = None
    sumRow['lastNonDetectionMJD'] = None
    for line in rf2.readlines():

      if("103.149" in line and lndSwitch ==0): #NON-DETECTION
        ndMJD = float(fl(reJD))-2400000.5
        ndDate = u.getDateFromMJD(ndMJD)
      elif("103.149" not in line and lndSwitch ==0): #FIRST REAL DETECTION
        # CATCH LAST NON-DETECTION
        sumRow['lastNonDetectionMJD'] = ndMJD
        sumRow['lastNonDetectionDate'] = ndDate
        lndSwitch = 1
        sumRow['observationMJD'] = float(fl(reJD)) - 2400000.5
        sumRow['observationDate'] = u.getDateFromMJD(sumRow['observationMJD'])
        sumRow['mag'] = fl(reMag)
      elif("103.149" not in line and lndSwitch == 1):
        mjd = float(fl(reJD)) - 2400000.5
        mag = fl(reMag)
        odate = u.getDateFromMJD(mjd)
        point = {'observationMJD':mjd,'mag':mag,'observationDate':odate}
        recRow.extend([point])

      print "lnd: ", sumRow['lastNonDetectionDate']
      print "lndmjd: ", sumRow['lastNonDetectionMJD']

    ## CHECK FOR OBJECTS WITH NO LAST NON-DETECTION
    if lndSwitch==0:
      sumRow['lastNonDetectionMJD'] = None
      sumRow['lastNonDetectionDate'] = None

    ## ADD EXTRA COLUMNS
    sumRow['dateCreated'] = b.get_now_sql_datetime()
    sumRow['dateLastModified'] = b.get_now_sql_datetime()
    sumRow['summaryRow'] = 1
    sumRow['ingested'] = 0
    sumRow['survey'] = "ogleIV"
    sumRow['surveyObjectUrl'] = "http://ogle.astrouw.edu.pl/ogle4/transients/transients.html"
    sumRow['filter'] = "I"
    ## KEYS TO APPEND TO RECURRENCE ROWS FROM SUMMARY ROW
    dups = ['name','raDeg','decDeg','dateCreated','dateLastModified','ingested','survey','filter']
    for point in recRow:
      for dup in dups:
        point[dup] = sumRow[dup]
        point['summaryRow'] = 0

    ## INGEST THE SUMMARY ROW
    try:
      log.debug("attempting to ingest the summary row for OGLE event")
      m.convert_dictionary_to_mysql_table(dbConn, sumRow, 'fs_ogle', ['observationMJD','name'])
    except Exception, e:
      log.critical("could not ingest the summary row for OGLE event - failed with this error %s: " % (str(e),))
      return -1

    ## INGEST THE RECURRANCE ROWS
    for point in recRow:
      try:
        log.debug("attempting to ingest the recurrance row for OGLE event")
        m.convert_dictionary_to_mysql_table(dbConn, point, 'fs_ogle', ['observationMJD','name'])
      except Exception, e:
        log.critical("could not ingest the recurrance row for OGLE event - failed with this error %s: " % (str(e),))
        return -1

    rf2.close()

  log.debug('we have %s ogle datablocks to parse for ogle objects' % (str(count),))

  rf.close()

  return None
