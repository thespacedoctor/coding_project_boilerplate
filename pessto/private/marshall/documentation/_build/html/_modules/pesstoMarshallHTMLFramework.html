      import dryxHTMLFramework as hf
      gh = lambda x: hf.get_html_block(x)
      gi = lambda x: hf.get_input_block(x)

      #### GENERATE FORM LABELS (DIVS) ####
      #--------------------------------------------------------------------------------#
      labelDict = {}
      for label in self.labelList:
        labelDict[label] = gh(

                              )

      #### GENERATE IMAGES (DIVS) ####
      #--------------------------------------------------------------------------------#

                        )
                      )

      #### GENERATE TEXTBOXES (INPUTS) ####
      #--------------------------------------------------------------------------------#
      textDict = {}
      for text in self.textboxList:
        textDict[text] = dict(
                                tag="input",
                                name=text.replace(' ',''),
                                type='text',
                                placeholder=text
                              )

      #### GERERATE SELECTS ####
      #--------------------------------------------------------------------------------#
      selectDict = {}
      for select in self.selectList:
        selectDict[select] = dict(tag="select",
                                    htmlClass=select+"Select",
                                    htmlId=select.replace(' ','')
                                  )

      #### GERERATE CHECKBOXES ####
      #--------------------------------------------------------------------------------#
      checkboxDict = {}
      for checkbox in self.checkboxList:
        checkboxDict[checkbox] = dict(
                                        tag="input",
                                        htmlId=checkbox.replace(' ',''),
                                        name=checkbox,
                                        value=checkbox,
                                        type='checkbox'
                                      )

      #### GENERATE BUTTONS ####
      #--------------------------------------------------------------------------------#
      buttonDict = {}
      for button in self.buttonList:
        buttonDict[button] = dict(
                                  tag="button",
                                  htmlClass="greyButton",
                                  htmlId=button.replace(' ','').replace('(','').replace(')',''),
                                  blockContent=button
                                )


      #### GENERATE THE ROWS ####
      #--------------------------------------------------------------------------------#
      i = 1
      rowList = []
      rowList.append("NULL")
      while(i <= self.numberOfRows):
        rowName = "row"+"{0:02.0f}".format(i)
        rowList.append(
                        dict(
                              tag="div",
                              htmlClass="divHorizontalKids",
                              htmlId=rowName
                        )
                      )
        i += 1

      form = dict(
                    tag="form",
                    htmlClass="floatingForm",
                    method="post"
                  )

      formContent = dict(
                          tag="div",
                          htmlClass="floatingFormContent"
                        )

      formWindow = dict(
                          tag="div",
                          htmlClass='floatingFormWindow'
                        )
      fader = dict(
                    tag="div",
                    htmlClass="fadeBackground"
                  )

      return pesstoIcon,labelDict,textDict,selectDict,checkboxDict,buttonDict,rowList,formContent,form,fader,formWindow

  ################ >ACTION(S) ################
  ## BLOCKS
  blockDict = {}

                                tag="div",
                                htmlClass='dropDownMenu',

                                  )

  ## BUTTONS
  blockDict['moveHover']["blockContent"] = 'move'

  for item in buttons:
    buttonDict[item] = dict(
                              tag="div",
                              htmlClass='moveMenuButton',
                              htmlId=item+'Button',
                              blockContent=item
                            )

    elif(item is 'archive'): imwfflag = 'archive'
    elif(item is 'obs queue'): imwfflag = 'pending observation'
    elif(item is 'follow'): imwfflag = 'following'
    elif(item is 'observed'): imwfflag = 'pending classification'
    elif(item is 'complete'): imwfflag = 'followup complete'
    elif(item is 'try again'): imwfflag = 'pending observation'

    buttonDict[item]["extraAttr"] = 'mwfFlag="'+imwfflag+'"'


  if(re.search('inbox', mwfFlag, re.IGNORECASE)):
    blockDict['moveMenu']["blockContent"] += gh(buttonDict['archive'])+\
                                              gh(buttonDict['obs queue'])

  elif(re.search('review for followup', mwfFlag, re.IGNORECASE)):
    blockDict['moveMenu']["blockContent"] += gh(buttonDict['archive']) +\
                                           gh(buttonDict['follow'])

  elif(re.search('archive', mwfFlag, re.IGNORECASE) and (cFlag == 0)):
  #elif(re.search('archive', mwfFlag, re.IGNORECASE)):

                                           gh(buttonDict['obs queue'])
