           sys.exit('error: the frame can not be corrected')




[docs]def mysqlCreateTableFromHeaderQuery(frame, dbTableName):
	# GRAB THE FITS HEADER
	header = grab_fits_header(frame)
	query = ""

	# CREATION PREAMBLE
	preamble = "SET @saved_cs_client = @@character_set_client;\nSET character_set_client = utf8;\nCREATE TABLE `"+dbTableName+"` (\n"

	# SOME STARTING PESSTO MARSHALL COLUMNS
	createTable = "`id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT 'PM: An internal counter',\n"
	createTable += "`transientBucketId` bigint(20) NOT NULL,\n"
	createTable += "`fileName` varchar(50) DEFAULT NULL COMMENT 'PM: FILENAME',\n"
	createTable += "`imageType` varchar(50) DEFAULT NULL COMMENT 'PM: FILETYPE',\n"

	# GRAB FITS KEYWORDS AND PRODUCE MYSQL COMMANDS WITH CORRECT COLUMN TYPES
	for key in header.keys():
		#STRIP FITS HEADER KEYWORDS OF WHITESPACE
		formatedKey = (key.replace(" ","_")).strip()
		#print formatedKey+":\t"+str(header[key])
		if(len(key)>0):
			createTable += "`"+formatedKey+"` "
			if(type(header[key]) == str and re.search('^[0-9]{4}-[0-9]{2}-[0-9]{2}T',header[key])):
				createTable += "datetime NOT NULL"
			elif(type(header[key]) == str and len(header[key]) < 30):
				createTable += " varchar(50) DEFAULT NULL"
			elif(type(header[key]) == str and len(header[key]) >= 30 and len(header[key]) < 80):
				createTable += " varchar(100) DEFAULT NULL"
			elif(type(header[key]) == str):
				createTable += " varchar(200) DEFAULT NULL"
			elif(type(header[key]) == int and abs(header[key]) <= 9):
				createTable += " tinyint DEFAULT NULL"
			elif(type(header[key]) == int):
				createTable += " int DEFAULT NULL"
			elif(type(header[key]) == float):
				createTable += " double DEFAULT NULL"
			elif(type(header[key]) == bool):
				createTable += " tinyint DEFAULT NULL"
			else:
				createTable += " ??????????????????????????????????"

			if(key is not formatedKey):
				createTable += " COMMENT 'original fits keyword: "+key+"',\n"
			else: createTable += ",\n"

			#createTable += "\t\t"+str(header[key])+"\n"

	# EXTRA PESSTO MARSHALL COLUMNS
	createTable += "`htm20ID` bigint(20) unsigned DEFAULT NULL COMMENT 'PM: Column required for efficient conesearching',\n"
	createTable += "`htm16ID` bigint(20) unsigned DEFAULT NULL COMMENT 'PM: Column required for efficient conesearching',\n"
	createTable += "`cx` double DEFAULT NULL COMMENT 'PM: Column required for efficient conesearching',\n"
	createTable += "`cy` double DEFAULT NULL COMMENT 'PM: Column required for efficient conesearching',\n"
	createTable += "`cz` double DEFAULT NULL COMMENT 'PM: Column required for efficient conesearching',\n"
	createTable += "`atQUB` tinyint DEFAULT 0 COMMENT 'PM: Flag to show if image in at QUB',\n"
	createTable += "`atTrieste` tinyint DEFAULT 0 COMMENT 'PM: Flag to show if image in at Trieste',\n"
	createTable += "`phaseIIIReleased` tinyint DEFAULT 0 COMMENT 'PM: Flag to show if image has been released to ESO as a PhaseIII product',\n"

	# EXTRA TABLE INFO
	createTable += "PRIMARY KEY (`id`),\n"
	createTable += "UNIQUE KEY `fileName_UNIQUE` (`fileName`)\n"

	# POSTAMBLE
	postamble = ") ENGINE=MyISAM AUTO_INCREMENT=0 DEFAULT CHARSET=latin1;\nSET character_set_client = @saved_cs_client;\n"

	query += preamble+createTable+postamble+"\n"
	return query
