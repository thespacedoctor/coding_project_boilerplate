    -

    - log -- logger

    - None

      """Method to ingest data from the feeder survey table(s) into the transientBucket

          Key Arguments:
            - dbConn -- mysql database connector

          Returns
            - None
      """
      ################ > IMPORTS ################
      import dryxMySQL as m

      ################ > SETTINGS VARIABLE ################
      dbConn=self.dbConn
      log=self.log
      # NAME OF THE VIEW WE ARE MATCHING AGAINST PRE-INGEST
      masterView = 'view_transientBucketMaster'
      cn = self.colNames

      ################ >ACTION(S) ################
      ## SET THE SUMMARY-ROW AND INGEST FLAGS FOR NEWLY ADDED ROWS ##

      try:
        log.debug("attempting to add summary and ingest flags to %s" % (self.basetable,))
        _add_ingested_and_summary_columns_to_tables(self.basetable,1,dbConn,log)
      except Exception, e:
        log.critical("could not add summary and ingest flags to %s - failed with this error %s: " % (self.basetable,str(e)))
        return -1

      ## ADD & POPULATE HTMID COLUMNS FOR ROWS MISSING THIS DATA NESSCESSARY FOR CROSS-MATCHING
      try:
        log.debug("attempting to add the HTMIds to %s" % (self.basetable,))
        m.add_HTMIds_to_mysql_tables(cn['raDeg'],cn['decDeg'],self.basetable,dbConn,log)
      except Exception, e:
        log.critical("could not add the HTMIds to %s - failed with this error %s: " % (self.basetable,str(e)))
        return -1

      ## GRAB ALL NEW ROWS
      try:
        log.debug("attempting to grab the new summary rows from the %s table" % (self.basetable,))
        newRows = m.execute_mysql_read_query(self.newRowsQuery,dbConn)
      except Exception, e:
        log.critical("could not grab the new summary rows from the %s table - failed with this error %s: " % (self.basetable,str(e)))
        return -1

      odate = False
      omjd = False
      if cn['observationDate']: odate = True
      if cn['observationMJD']: omjd = True

      ## ITERATE THROUGH THESE NEW ROWS, CROSSMATCHING AGAINST THE TRANSIENTBUCKET OBJECTS
      for row in newRows:
        # GET THE NAME FROM OTHER COLUMN IF NOT POPULATED
        if not row[cn["name"]]:
          self.colNames["name"] = self.colNames["name_other"]

        # CLEAN COLUMN VALUES
        d = _row_dictionary_cleaner(dbConn,log,row,columnNames=cn,mjdColumn=omjd,dateColumn=odate)

        # CONESEARCH THE TRANSIENTBUCKET OBJECTS (WITH MASTER-FLAG SET) TO FIND ANY MATCHING OBJECT ALREADY IN THE DATABASE
        try:
          log.debug("attempting to perform a conesearch on the transientBucket master view")
          message, transientBucketXMObjects = coneSearch(float(row[cn['raDeg']]),float(row[cn['decDeg']]), self.csRadius, masterView, queryType = FULL, conn = dbConn)
        except Exception, e:
          log.critical("could not perform a conesearch on the transientBucket master view - failed with this error %s: " % (str(e),))
          return -1
        if message and (message.startswith('Error') or 'not recognised' in message):
           # SUCCESSFUL CONE SEARCHES SHOULD NOT RETURN AN ERROR MESSAGE, OTHERWISE SOMETHING WENT WRONG.
           log.critical("could not perform a conesearch on the transientBucket master view - failed with this error %s: " % (message,))
           return -1

        ## SET THE DEFAULT VALUES HERE
        d['transientBucketId'] = -1
        d['masterIDFlag'] = 1

        ## IF THERE WAS A MATCH WITHIN THE TRANSIENTBUCKET THEN CHANGE VALUES FOR transientBucketId AND masterIDFlag FROM DEFAULT VALUES
        if transientBucketXMObjects:

           # WE HAVE A MATCH - AND WE ONLY WANT THE NEAREST ROW.
           d['masterIDFlag'] = 0
           numberOfMatches = len(transientBucketXMObjects)
           nearestSep = transientBucketXMObjects[0][0]
           nearestCatRow = transientBucketXMObjects[0][1]
           nearestCatId = nearestCatRow[CAT_ID_RA_DEC_COLS[masterView][0][0]]

           d['transientBucketId'] = nearestCatRow["transientBucketId"]
           #sys.exit(d['transientBucketId'])

        try:
          log.debug("attempting to ingest row into transientbucket from %s" % (self.basetable,))
          m.convert_dictionary_to_mysql_table(dbConn, d, 'transientBucket', ['name','observationMJD','magnitude','filter'])
          primaryKeyId = dbConn.insert_id()
          print d
          log.debug('primary key: %s' % (primaryKeyId,))
        except Exception, e:
          log.error("could not ingest row into transientbucket from %s - failed with this error %s: " % (self.basetable,str(e)))


        # SET MASTER FLAG IF THIS IS THE FIRST OBSERVATION OF AN OBJECT
        if d['masterIDFlag'] and primaryKeyId:
           #print d['masterIDFlag'], primaryKeyId
           _update_transientBucketId_for_master_event(dbConn, primaryKeyId)
           d['transientBucketId'] = primaryKeyId
           d['masterIDFlag'] = 0

        # SWITCH THE INGESTED FLAG TO TRUE
        if primaryKeyId:
          sqlQuery = """UPDATE """+self.basetable+"""
                          set ingested = 1
                          where primaryid = %s""" % (row['primaryId'],)

          m.execute_mysql_write_query(sqlQuery,dbConn)

          if row.has_key('circularComment'):
            sqlQuery = "INSERT into pesstoObjectsComments(pesstoObjectsId,commentAuthor,dateLastModified,dateCreated,comment) values('%s','CRTS Circular','%s','%s','%s')" % (primaryKeyId,row["circularDate"],row["circularDate"],row['circularComment'])
            print sqlQuery
            m.execute_mysql_write_query(sqlQuery,dbConn)

        ## GRAB ALL NEW RECURRENCE ROWS
        if(self.recRowsQuery):
          try:
            log.debug("attempting to grab the new recurrence rows from the %s table" % (self.basetable,))
            sqlQuery = self.recRowsQuery+" and name = \""+d['name']+"\""
            recRows = m.execute_mysql_read_query(sqlQuery,dbConn)
          except Exception, e:
            log.critical("could not grab the new recurrence rows from the %s table - failed with this error %s: " % (self.basetable,str(e)))
            return -1

          for recRow in recRows:
            # CLEAN COLUMN VALUES
            rd = _row_dictionary_cleaner(dbConn,log,recRow,columnNames=cn,mjdColumn=omjd,dateColumn=odate)
            #print rd

            rd['masterIDFlag'] = 0
            rd['transientBucketId'] = d['transientBucketId']
            print rd['transientBucketId']

            try:
              log.debug("attempting to ingest row into transientbucket from %s" % (self.basetable,))
              m.convert_dictionary_to_mysql_table(dbConn, rd, 'transientBucket', ['name','observationMJD','magnitude','filter'])
              primaryKeyId = dbConn.insert_id()
              log.debug('primary key: %s' % (primaryKeyId,))
            except Exception, e:
              log.error("could not ingest row into transientbucket from %s - failed with this error %s: " % (self.basetable,str(e)))

            # SWITCH THE INGESTED FLAG TO TRUE
            if primaryKeyId:
              sqlQuery = """UPDATE """+self.basetable+"""
                              set ingested = 1
                              where primaryid = %s""" % (recRow['primaryId'],)

            m.execute_mysql_write_query(sqlQuery,dbConn)

      return None

[docs]def _update_transientBucketId_for_master_event(dbConn, primaryKeyId):
   """
   Sets the ``transientBucketId`` to the ``primaryKeyId`` in the ``transientBucket`` db table (for the object with the ``masterIDFlag`` set).
   """
   import MySQLdb
   rowsUpdated = 0

   try:
      cursor = dbConn.cursor(MySQLdb.cursors.DictCursor)

      cursor.execute ("""
          update transientBucket
          set transientBucketId = %s
          where primaryKeyId = %s
          and masterIDFlag is true
      """, (primaryKeyId, primaryKeyId))

      rowsUpdated = cursor.rowcount

      # Did we update any transient object rows? If not issue a warning.
      if rowsUpdated == 0:
         print "WARNING: No transient object entries were updated."

      cursor.close ()


   except MySQLdb.Error, e:
      print "Error %d: %s" % (e.args[0], e.args[1])

   return rowsUpdated

  """
  Clean up some of the columns before ingest

  Key Arguments:
    - row -- a dictionary of the db column keys and values for one db row
    - dbConn -- mysql database connection
    - log -- logger
    - columnNames -- a dictionary of all the db column names
    - mjdColumn -- name of the column containing an observation mjd
    - dateColumn -- name of the column containing an observation date
    - surveyName -- name of the column containing the survey name

  Returns
    - d -- the cleaned up row dictionary
  """
  ################ > IMPORTS ################
  import datetime
  import re

  ################ >SETTINGS ################
  currentDate = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")

  ################ >ACTION(S) ################
  # INITIATE THE INGEST DICTIONARY
  d = {}
  for key, value in columnNames.iteritems():
    if value:
      d[key] = row[value]
    else:
      d[key] = None

  ## CLEAN PREDICTION SOURCE
  if d["transientTypePredicationSource"] and "crts" in d["transientTypePredicationSource"]:
    d["transientTypePredicationSource"] = "CRTS Circular"

  ## CLEAN PREDICTIONS
  if d["transientTypePrediction"] and len(d["transientTypePrediction"]) > 30:
    d["transientTypePrediction"] = "Supernova or Other"

  ## CONVERT FROM JD TO MJD
  if mjdColumn and 2450000 < d['observationMJD'] < 2500000:
    d['observationMJD'] = d['observationMJD'] -2400000.5

  ## SOME DATES ARE SCREWED UP IN THE VOEVENTS - FORCE THEM THROUGH AND MANUALLY FIX LATER
  if dateColumn and not d['observationDate']:
    d['observationDate'] = "1981-05-18 12:00:00"
  if mjdColumn and not d['observationMJD']:
    d['observationMJD'] = u.getMJDFromSqlDate(strip(d['observationDate']))

  if not dateColumn:
    d['observationDate'] = u.getDateFromMJD(row[cn['observationMJD']])
  elif not mjdColumn:
    d['observationMJD'] = u.getMJDFromSqlDate(str(row[cn['observationDate']]))

  ## GRAB FILTER FROM VOEVENT SYNTAX
  if 'phot.mag;em.opt' in d['filter']:
    d['filter'] = re.search('opt\.(.*$)',d['filter']).group(1)

  ## ADD EXTRA ITEMS TO THE INGEST DICTIONARY
  d['dateCreated'] = currentDate
  d['dateLastModified'] = currentDate
  d['htm20ID'] = row['htm20ID']
  d['htm16ID'] =  row['htm16ID']
  d['cx'] =  row['cx']
  d['cy'] =  row['cy']
  d['cz'] =  row['cz']

  ## OVERRIDES cn['survey'] IF surveyName IS GIVEN
  if surveyName:
    d['survey'] = surveyName

  return d

[docs]def _add_ingested_and_summary_columns_to_tables(tableName,summaryDefault,dbConn,log):
  """
  Function to add 'ingested' and 'summary' flag columns to mysql db table

  Key Arguments:
    - tableName -- name of table to columns to
    - dbConn -- database hosting the above table
    - log -- logger

  Returns
    - None
  """
  ## > IMPORTS ##
  import dryxMySQL as m

  ## >SETTINGS ##

  # >ACTION(S)  ###
  colsToAdd = {'summaryRow':'TINYINT','ingested':'TINYINT'}

  # CHECK IF COLUMN EXISTS YET - IF NOT CREATE IT
  for key in colsToAdd.keys():
    log.debug("attempting to check and generate the ``ingested`` & ``summaryRow`` columns for the %s db table" % (tableName,))
    try:
      colExists = """SELECT *
                  FROM information_schema.COLUMNS
                  WHERE TABLE_SCHEMA=DATABASE()
                  AND COLUMN_NAME='%s'
                  AND TABLE_NAME='%s'""" % (key,tableName,)
      colExists = m.execute_mysql_read_query(colExists,dbConn)
      if not colExists:
          sqlQuery = "ALTER TABLE "+tableName+" ADD "+key+" "+htmCols[key]+" DEFAULT NULL"
          m.execute_mysql_write_query(sqlQuery,dbConn)
    except Exception, e:
      log.critical("could not check and generate the ``ingested`` & ``summaryRow`` columns for the %s db table - failed with this error %s: " % (tableName,str(e),))
      return -1

  sqlQuery = "SELECT primaryId from "+tableName+" where summaryRow is NULL"
  rows = m.execute_mysql_read_query(sqlQuery,dbConn)

  for row in rows:
    sqlQuery = """UPDATE %s SET summaryRow=%s
                        where primaryId = '%s'""" % (tableName,summaryDefault,row['primaryId'])
    try:
      log.debug("attempting to set the default values for the summaryRow column of the %s db table" % (tableName,))
      m.execute_mysql_write_query(sqlQuery,dbConn)
    except Exception, e:
      log.error("could not set the default values for the summaryRow column of the %s db table - failed with this error %s: " % (tableName,str(e),))
      return -1


  sqlQuery = "SELECT primaryId from "+tableName+" where ingested is NULL"
  rows = m.execute_mysql_read_query(sqlQuery,dbConn)

  for row in rows:
    sqlQuery = """UPDATE %s SET ingested=0
                        where primaryId = '%s'""" % (tableName,row['primaryId'])
    try:
      log.debug("attempting to set the default values for the ingested column of the %s db table" % (tableName,))
      m.execute_mysql_write_query(sqlQuery,dbConn)
    except Exception, e:
      log.error("could not set the default values for the ingested column of the %s db table - failed with this error %s: " % (tableName,str(e),))
      return -1

  return None
