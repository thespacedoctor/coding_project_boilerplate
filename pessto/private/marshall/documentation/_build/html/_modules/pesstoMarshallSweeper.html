[docs]def retrospectively_set_lsq_non_detection_dates(dbConn,log):
  """LSQ often set a last non-detection date for an object after the release of the discovery information.
  This function looks back at the LSQ date to find updated last non-detection dates and propagates the information into the marshall.

  Key Arguments:
    - dbConn -- mysql database connection
    - log -- logger

  Returns
    - None
  """
  ################ > IMPORTS ################
  import dryxMySQL as m

  ################ > VARIABLE SETTINGS ######

  ################ >ACTION(S) ################
  ########## FIND THE LSQ LAST NON-DETECTIONS ##########
  sqlQuery = """SELECT s.candidateId,s.lastNonDetection
                from fs_lsq_current_summaries s, fs_lsq l
                where s.candidateId = l.candidateId and l.summaryRow = 1 and
                  l.lastNonDetection != s.lastNonDetection and s.lastNonDetection is not NULL"""
  try:
    log.debug("attempting to find the LSQ objects with updated last non-detection dates")
    rows = m.execute_mysql_read_query(sqlQuery,dbConn)
  except Exception, e:
    log.error("could not find the LSQ objects with updated last non-detection dates - failed with this error %s: " % (str(e),))
    return -1

  ## UPDATE THE fs_lsq DB TABLE
  for row in rows:
    try:
      sqlDate = datetime.datetime.strptime(row["lastNonDetection"],'%Y-%m-%d')
    except:
      sqlDate = row["lastNonDetection"]
    sqlQuery = """UPDATE fs_lsq
                  set lastNonDetection = %s
                  where candidateId = '%s'""" % (sqlDate,row["candidateId"],)
    try:
      log.debug("attempting to update objects in the ``fs_lsq`` table with new last non-detection dates")
      m.execute_mysql_write_query(sqlQuery,dbConn)
    except Exception, e:
      log.error("could not update objects in the ``fs_lsq`` table with new last non-detection dates - failed with this error %s: " % (str(e),))
      return -1


  sqlQuery = """SELECT DISTINCT t.transientBucketId, l.lastNonDetection
                from transientBucket t, fs_lsq_current_summaries l
                where t.name = l.candidateId and t.lastNonDetectionDate is NULL and l.lastNonDetection is not NULL"""

  try:
    log.debug("attempting to find lsq objects with new last non-detection dates set that are not set in the ``transientBucket`` table")
    rows = m.execute_mysql_read_query(sqlQuery,dbConn)
  except Exception, e:
    log.error("could not find lsq objects with new last non-detection dates set that are not set in the ``transientBucket`` table - failed with this error %s: " % (str(e),))
    return -1

  for row in rows:
    row["lastNonDetection"] = datetime.datetime.strptime(str(row["lastNonDetection"]),'%Y-%m-%d')
    sqlQuery = """UPDATE transientBucket
                  set lastNonDetectionDate = '%s', dateLastModified = NOW()
                  where transientBucketId = '%s'""" % (row["lastNonDetection"],row["transientBucketId"],)
    try:
      log.debug("attempting to update transientBucketId = %s with a new lsq last non-detection date = %s" % (str(row["lastNonDetection"]),str(row["transientBucketId"]),))
      m.execute_mysql_write_query(sqlQuery,dbConn)
    except Exception, e:
      log.error("could not update transientBucketId = %s with a new last non-detection date = %s - failed with this error %s: " % (str(e),))
      return -1

  return None

[docs]def update_sdss_coverage_flags(dbConn,log):
  """Update all new objects in the ``pesstoObjects`` table with the sdss_coverage flag.

  Key Arguments:
    - dbConn -- mysql database connection
    - log -- logger

  Returns
    - None
  """
  ################ > IMPORTS ################
  import dryxMySQL as m
  import time
  import pesstoMarshallCommonUtils as p
  import dryxAstroTools as at

  ################ > VARIABLE SETTINGS ######

  ################ >ACTION(S) ################
  ########## SET SDSS COVERAGE FLAG IN pesstoObjects TABLE ##########
  sqlQuery = "SELECT t.transientBucketId, t.raDeg, t.decDeg from pesstoObjects p, transientBucket t where p.sdss_coverage is NULL and p.transientBucketId = t.transientBucketId and t.masterIDFlag = 1"
  newRows = m.execute_mysql_read_query(sqlQuery,dbConn)

  for newRow in newRows:
    time.sleep(1)
    try:
      log.debug("attempting to check sdss coverage for ra & dec %s, %s" % (newRow['raDeg'], newRow['decDeg'],))
      sdss = at.check_for_sdss_coverage(dbConn, log, newRow['raDeg'], newRow['decDeg'])
    except Exception, e:
      log.error("could not check sdss coverage for transientBucketId - failed with this error %s: " % (str(e),))
      return -1

    if sdss:
      sqlQuery = "UPDATE pesstoObjects set sdss_coverage = %s where transientBucketId = %s" % (1,str(newRow['transientBucketId']),)
      try:
        log.debug("attempting to write sdss_coverage flag into db for transientBucketId %s" % (str(newRow['transientBucketId']),))
        m.execute_mysql_write_query(sqlQuery,dbConn)
      except Exception, e:
        log.error("could not write sdss_coverage flag into db for transientBucketId %s - failed with this error %s: " % (str(newRow['transientBucketId']),str(e),))
        return -1
    else:
      sqlQuery = "UPDATE pesstoObjects set sdss_coverage = %s where transientBucketId = %s" % (0,str(newRow['transientBucketId']),)
      try:
        log.debug("attempting to write sdss_coverage flag into db for transientBucketId %s" % (str(newRow['transientBucketId']),))
        m.execute_mysql_write_query(sqlQuery,dbConn)
      except Exception, e:
        log.error("could not write sdss_coverage flag into db for transientBucketId %s - failed with this error %s: " % (str(newRow['transientBucketId']),str(e),))
        return -1

  return None

[docs]def add_new_transients_to_pesstoObjects_table(dbConn,log):
  """Read the ``transientBucket`` table for new transients and add them to the ``pesstoObejcts`` db table.

  Key Arguments:
    - dbConn -- mysql database connection
    - log -- logger

  Returns
    - None
  """
  ################ > IMPORTS ################
  import dryxMySQL as m
  import time
  import pesstoMarshallCommonUtils as p

  ################ > VARIABLE SETTINGS ######

  ################ >ACTION(S) ################
  #### ADD NEW OBJECTS TO THE PESSTOOBJECTS TABLE ####
  #---------------------------------------------------------------------------------------------------------------------------------#
  # FIND OBJECTS THAT ARE NOT IN THE TRANSIENT BUCKET BUT NOT IN THE PESSTOOBJECTS TABLE
  sqlQuery = """SELECT DISTINCT transientBucketId
                FROM transientBucket
                WHERE transientBucketId NOT IN
                (
                SELECT transientBucketId
                FROM pesstoObjects
                )"""

  try:
    log.debug("attempting to find objects in the ``transientBucket`` table but not in the ``pesstoObjects`` table")
    rows = m.execute_mysql_read_query(sqlQuery,dbConn)
  except Exception, e:
    log.error("could not find objects in the ``transientBucket`` table but not in the ``pesstoObjects`` table - failed with this error %s: " % (str(e),))
    return -1

  for row in rows:
    tbId = row['transientBucketId']

    # ADD NEW OBJECTS INTO PESSTOOBJECTS TABLE
    sqlQuery ="""INSERT INTO pesstoObjects (
                  pesstoObjectsId,
                  transientBucketId,
                  classifiedFlag,
                  marshallWorkflowLocation,
                  alertWorkflowLocation,
                  publicStatus,
                  dateAdded,
                  dateLastModified)
                VALUES (%s, %s, 0,'Inbox','Pending Classification',1,NOW(),NOW())
              """ % (tbId, tbId,)

    try:
      log.info("attempting to add transient with transientBucketId = %s to the pesstoObjects db table" % (tbId,))
      m.execute_mysql_write_query(sqlQuery,dbConn)
      p.insert_changelog_entry(dbConn, log, tbId, 'object added to the pessto marshall inbox')
    except Exception, e:
      log.error("could not add transient with transientBucketId = %s to the pesstoObjects db table - failed with this error %s: " % (tbId,str(e),))
      return -1

  return None

  #---------------------------------------------------------------------------------------------------------------------------------#
  sqlQuery = """SELECT DISTINCT transientBucketId, survey
                FROM transientBucket WHERE
                (
                  spectralType is not NULL
                  AND transientBucketId NOT IN
                  (
                    SELECT transientBucketId
                    FROM pesstoObjects
                    WHERE classifiedFlag = 1
                  )
                )
            """

  try:
    log.debug("attempting to query for objects in ``transientBucket`` table that now have an external classification")
    rows = m.execute_mysql_read_query(sqlQuery,dbConn)
  except Exception, e:
    log.error("could not query for objects in transientBucket table that now have an external classification - failed with this error %s: " % (str(e),))
    return -1

  for row in rows:
    tbId = row['transientBucketId']
    survey = row['survey']
    print str(tbId)+" "+str(survey)+" \n"

    # CHANGE CLASSIFICATION FLAG, MWFFLAG AND AWFFLAG

    # UPDATE CHANGELOG TABLE
    p.insert_changelog_entry(dbConn, log, tbId, 'object classified by an external source (not pessto)')

  return None

  """Flag all non-*supernova* and non-*other* CRTS objects as ingested (i.e. no more work to be done with these objects)

  Key Arguments:
    - dbConn -- mysql database connection
    - log -- logger

  Returns
    - None
  """
  ################ > IMPORTS ################
  import dryxMySQL as m

  ################ > VARIABLE SETTINGS ######

  ################ >ACTION(S) ################
  sqlQuery = """SELECT primaryId FROM rss_channel_crts WHERE ingested = 0 and objectType not like "%other%" and objectType not like "%supernova%" and objectType is not Null"""
  try:
    log.debug("attempting to read CRTS objects that are not 'supernova' or 'other'")
    IDs = m.execute_mysql_read_query(sqlQuery,dbConn)
  except Exception, e:
    log.error("could not read CRTS objects that are not supernova or other - failed with this error %s: " % (str(e),))
    return -1

  try:
    log.debug("attempting to flag non-supernova and non-other CRTS objects as ingested")
    for i in IDs:
      m.set_flag(dbConn,"rss_channel_crts","ingested",1, "primaryId", i["primaryId"])
  except Exception, e:
    log.error("could not flag non-'supernova' and non-'other' CRTS objects as ingested - failed with this error %s: " % (str(e),))
    return -1

  return None

      UPDATE pesstoObjects
      SET classifiedFlag = 1

      UPDATE pesstoObjects
