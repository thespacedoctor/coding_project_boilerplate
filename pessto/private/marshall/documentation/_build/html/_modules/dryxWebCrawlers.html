[docs]def _fetch(url):
    import logging as log
    from eventlet.green import urllib2

    try:
      log.info('downloading '+url)
      body = urllib2.urlopen(url).read()
    except Exception, e:
      log.error("could not download "+url+" : "+str(e)+"\n")
      url = None
      body = None

    return url, body

[docs]def multiWebDocumentDownloader(urlList,downloadDirectory,log,dbConn,timeStamp=1,timeout=180,concurrentDownloads=10):
  """Given a list of document URLs this function will download the documents to a specified download directory

      Key Arguments:
        - urlList -- list of document urls
        - downloadDirectory -- directory to download the documents to

      Returns
        - list of timestamped documents (same order as the input urlList)
  """
  ## > IMPORTS ##
  import sys
  import os
  import eventlet
  from eventlet.green import urllib2
  import dryxCommonUtils
  import socket
  import re

  ## >SETTINGS ##
  # timeout in seconds
  socket.setdefaulttimeout(timeout)

  ###########################################################
  # >ACTION(S)                                              #
  ###########################################################
  ## BUILD THE 2D ARRAY FOR MULTI_THREADED DOWNLOADS
  thisArray = []
  bodies = []
  localUrls = []
  theseUrls = []
  for url in urlList:
    filename = dryxCommonUtils.extractFilenameFromURL(url)    # EXTRACT THE FILENAME FROM THE URL
    if(timeStamp):
      filename = dryxCommonUtils.appendNowDateStampToFilename(filename)    # APPEND TIMESTAMP TO THE FILENAME
    localFilepath = downloadDirectory+"/"+filename    # GENERATE THE LOCAL FILE URL
    thisArray.extend([[url,localFilepath]])

  pool = eventlet.GreenPool(concurrentDownloads)
  i = 0
  try:
    log.debug("starting mutli-threaded download batch (%s) concurrent downloads")
    for url, body in pool.imap(_fetch, urlList):
      if(body):
        bodies.extend([body])
        theseUrls.extend([thisArray[i][1]])
      else:
        theseUrls.extend([None])
        bodies.extend([None])

      ## DUMP THE FILES FROM MEMORY EVERY CONCURRENT DOWNLOAD CYCLE
      if i%concurrentDownloads == 0:
        _dump_files_to_local_drive(bodies,theseUrls,dbConn,log)
        localUrls.extend(theseUrls)
        ## RESET THE TMP ARRAYS
        bodies = []
        theseUrls = []
      i += 1
  except eventlet.Timeout, e:
    log.error("something went wrong with the mutli-threaded download : "+str(e)+"\n")

  ## DUMP REMAINING FILES TO THE LOCAL DRIVE
  _dump_files_to_local_drive(bodies,theseUrls,dbConn,log)

  return localUrls

  """

    filepath = localUrlList[0]
  except Exception, e:
    myError = "could not convert single url to list and downloading : "+str(e)+"\n"
    log.error(myError)

  #log.debug('>>>>>>>>>>>>>>> the local url is '+ filepath)
