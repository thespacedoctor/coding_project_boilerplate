[docs]	def get_objects(self):

		#### GENERATE FORM LABELS (DIVS) ####
		#--------------------------------------------------------------------------------#
		labelDict = {}
		for label in self.labelList:
			labelDict[label] = hf.html_div_class()
			labelDict[label].htmlId = label.replace(' ','').replace(':','')
			labelDict[label].htmlClass = "labels"
			labelDict[label].blockContent = label

		#### GENERATE IMAGES (DIVS) ####
		#--------------------------------------------------------------------------------#
		pesstoIcon = hf.html_img_class()
		pesstoIcon.htmlId = 'formPesstoIcon'
		pesstoIcon.htmlClass = 'icon'
		pesstoIcon.src = '/pessto/images/pessto_icon.png'
		pesstoIcon.alt = 'pessto icon'

		#### GENERATE TEXTBOXES (INPUTS) ####
		#--------------------------------------------------------------------------------#
		textDict = {}
		for text in self.textboxList:
			textDict[text] = hf.html_input_class()
			textDict[text].name = text.replace(' ','')
			textDict[text].htmlClass = ""
			textDict[text].ttype = 'text'
			textDict[text].placeholder = text
			textDict[text].blockContent = ""

		#### GERERATE SELECTS ####
		#--------------------------------------------------------------------------------#
		selectDict = {}
		for select in self.selectList:
			selectDict[select] = hf.html_select_class()
			selectDict[select].htmlId = select.replace(' ','')
			selectDict[select].htmlClass = select+"Select"
			selectDict[select].blockContent = ""

		#### GERERATE CHECKBOXES ####
		#--------------------------------------------------------------------------------#
		checkboxDict = {}
		for checkbox in self.checkboxList:
			checkboxDict[checkbox] = hf.html_input_class()
			checkboxDict[checkbox].ttype = 'checkbox'
			checkboxDict[checkbox].name = checkbox
			checkboxDict[checkbox].value = checkbox
			checkboxDict[checkbox].htmlId = checkbox.replace(' ','')
			checkboxDict[checkbox].htmlClass = ""

		#### GENERATE BUTTONS ####
		#--------------------------------------------------------------------------------#
		buttonDict = {}
		for button in self.buttonList:
			buttonDict[button] = hf.html_button_class()
			buttonDict[button].htmlId = button.replace(' ','').replace('(','').replace(')','')
			buttonDict[button].htmlClass = "greyButton"
			buttonDict[button].blockContent = button

		#### GENERATE THE ROWS ####
		#--------------------------------------------------------------------------------#
		i = 1
		rowList = []
		rowList.append("NULL")
		while(i <= self.numberOfRows):
			rowName = "row"+"{0:02.0f}".format(i)
			rowList.append(hf.html_div_table_hk_cell_class())
			rowList[i].htmlId = rowName
			i += 1

		form = hf.HTMLDivForm()
		form.htmlClass = "floatingForm"
		formContent = hf.html_div_table_vk_cell_class()
		formContent.htmlClass = "floatingFormContent"
		formWindow = html_overlay_window_class()
		formWindow.htmlClass = 'floatingFormWindow'
		fader = html_fade_background_class()

		return pesstoIcon,labelDict,textDict,selectDict,checkboxDict,buttonDict,rowList,formContent,form,fader,formWindow
