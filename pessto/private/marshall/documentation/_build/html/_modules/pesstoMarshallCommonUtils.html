  """
  Create a connector to the pessto marshall database & setip logging
  Can determine from the path if a connection to the marshall or snadbox marshall is to be made.

    Key Arguments:
      - None

    Returns
      - dbConn - database connection
      - log - logger
  """
  ################ > IMPORTS ################
  import os
  import dryxMySQL as m
  import dryxLogging as l

  ################ >SETTINGS ################
  path = os.getcwd()

  ################ >ACTION(S) ################
  ## READ THE PATH OF THIS MODULE TO - SANDBOX OR MARSHALL?
  if("sandbox" in path):
    dbConn = m.set_db_connection('/pessto_marshall_sandbox/settings/database_credentials.yaml')
    log = l.setup_dryx_logging('/pessto_marshall_sandbox/settings/logging.yaml')
  else:
    dbConn = m.set_db_connection('/pessto_marshall/settings/database_credentials.yaml')
    log = l.setup_dryx_logging('/pessto_marshall/settings/logging.yaml')

  return dbConn, log
