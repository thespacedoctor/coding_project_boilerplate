[docs]def nullValue(value):
   returnValue = None

   if value and value.strip():
      returnValue = value.strip()

   return returnValue




[docs]def floatValue(value):
   returnValue = None

   if value:
      try:
         returnValue = float(value)
      except ValueError, e:
         pass

   return returnValue

   return returnValue
