  """Used simply to set the python path for the pessto marshall scripts - Apache pythonpath not the same as the users path.

  Key Arguments:
    - None

  Returns
    - None
  """
  ################ > IMPORTS ################
  import yaml

  ## IMPORT THE YAML PYTHONPATH DICTIONARY ##
  path = os.getcwd()

  ################ >ACTION(S) ################
  ## READ THE PATH OF THIS MODULE TO - SANDBOX OR MARSHALL?
  if("sandbox" in path):
    pathToYamlFile = '/pessto_marshall_sandbox/settings/python_path.yaml'
  else:
    pathToYamlFile = '/pessto_marshall/settings/python_path.yaml'

  try:
    stream = file(pathToYamlFile, 'r')
    ppDict = yaml.load(stream)
  except Exception, e:
    print str(e)

  # READ THE KEYS FROM THE YAML DICTIONARY AND APPEND TO PYTHONPATH
  svnroot = ppDict["svnroot"]
  pythonpaths = ppDict["pythonpath"]
  for key in pythonpaths:
    sys.path.append(svnroot+pythonpaths[key])

  return



if __name__ == '__main__':
  main()
